            event.preventDefault();
            var username = document.getElementById('instagramUsername').value;
            // Validar el nombre de usuario de Instagram
            if (username) {
                // Generar la clave del WiFi 
                var wifiKey = generarClaveWiFi();
                // Mostrar la clave del WiFi
                document.getElementById('wifiKeyText').textContent = wifiKey;
                document.getElementById('wifiKey').classList.remove('hidden');
            } else {
                // Mostrar un mensaje de error si no se proporciona un nombre de usuario de Instagram
                document.getElementById('errorMessage').textContent = 'Por favor, introduce tu nombre de usuario de Instagram.';
                document.getElementById('errorMessage').classList.remove('hidden');
            }
        });
